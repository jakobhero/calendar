#show a user's data
curl http://127.0.0.1:5000/user/jakob

#add a new calendar
curl -X POST http://127.0.0.1:5000/user/jakob --data "calendar=New Calendar"

#to get all appointments
curl http://127.0.0.1:5000/appointments/jakob/test_calendar

#to add a new appointment
curl http://127.0.0.1:5000/appointments/jakob/test_calendar --data "name=Test Meeting&start=0&dur=60"

#to update an appointment (appointment with same name and date must exist in calendar)
curl http://127.0.0.1:5000/appointments/jakob/test_calendar --data "name=Test Meeting&start=1000&dur=30"

#to delete an appointment (appointment with same name and exact same timestamp must exist in calendar)
curl -X DELETE http://127.0.0.1:5000/appointments/jakob/test_calendar --data "app_name=New Meeting&app_start=3723"
